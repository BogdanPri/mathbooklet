    \coordinate (A) at (0,3);
    \coordinate (B) at (0,2);
    \coordinate (C) at (0,1);

    \coordinate (D) at (4,3);
    \coordinate (E) at (4,2);

    \foreach \x/\y in {A/D, B/D, C/F}{%
        \draw[thick,->] ($(\x) + (0.25,0)$) -- ($(\y) + (-0.25,0)$);
    }
